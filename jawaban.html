1. How to create ajax get to retrieve banners data from server, that has error handling below: - Internal server error - Request timed out - No data available GET /api/banners?country=SG response: [ { data:[ {object}, {object}, {object} ] } ]

Answer : 

$('#test').on('click', function(){
                        $.ajax({
                            dataType: 'json',
                                type:"GET",
                                    url:"/api/banners?country=SG ",
                                        success: function(response, data) {
                                            console.log(data.data);
                                            },
                                                error: function(response, xhr,error) {
                                                    console.log(error) 
  }
                                                    });
                                                    });
